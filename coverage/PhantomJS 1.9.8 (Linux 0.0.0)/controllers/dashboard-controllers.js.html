function changeChartColor($scope, percentage, index) {
    // 60%, 85%, 100% 구분해서 색상 표시
    switch(true) {
        case (percentage < 60): $scope.chartColor[index] = '#49ad3a'; break;
        case (percentage >= 60 && percentage < 85): $scope.chartColor[index] = '#f3a952'; break;
        default: $scope.chartColor[index] = '#ca2020'; break;
    }
}
