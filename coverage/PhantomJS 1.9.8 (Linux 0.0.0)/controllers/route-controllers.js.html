'use strict';
 
var routeControllers = angular.module('route.controllers', []);
 
routeControllers.controller('RouteStateController', function ($rootScope, $scope, cloudfoundry, routeServices, commonUtil, $translate) {
 
    $scope.titles = commonUtil.getTitleLinks();
    $scope.tenantTitles = commonUtil.getTenantTitles();
 
	var currentPage = 1;
	var itemsPerPage = 10;
    var hosts = [];
 
    $scope.getList = function (obj) {
        var condition = '';
 
        if(obj == undefined) {
            obj = {"currentPage":$scope.currentPage, "itemsPerPage":$scope.itemsPerPage};
        }
        var requestPage = obj.currentPage-1;
 
        if(($scope.searchField == undefined || $scope.searchField == '') && jQuery('#searchText').text() != '' && jQuery('#searchText').text().trim() != $translate.instant('host_name_search')) {
            cloudfoundry.alert($translate.instant('mi_search_predictive_func'));
            return;
        }
        if($scope.searchField != undefined && (typeof $scope.searchField == 'object' && $scope.searchField.length > 0)) {
            $scope.currentPage = 1;
            obj.currentPage = 1;
            requestPage = 0;
            condition = "host:" + $scope.searchField[0].name;
        }
 
        $rootScope.loading = true;
        var routePromise = routeServices.listRoutes(obj.itemsPerPage, requestPage, condition);
        routePromise.success(function (data, status, headers) {
            $rootScope.loading = false;
 
            $scope.totalItems = data.totalElements;
            $scope.currentPage = obj.currentPage;
            $scope.itemsPerPage = obj.itemsPerPage;
            $scope.totalPages = Math.ceil(data.totalElements / obj.itemsPerPage);
 
            $scope.routes = data.content;
        });
        routePromise.error(function (data) {
            $rootScope.loading = false;
            cloudfoundry.alert(data);
        });
    }
 
    $scope.getListNames = function () {
        var routePromise = routeServices.listAllRoutes();
        routePromise.success(function(data) {
            angular.forEach(data, function(host) {
                var item = host.host;
                if(hosts.indexOf(item) == -1)
                    hosts.push(item);
            });
        })
        routePromise.error(function (data) {
            cloudfoundry.alert(data);
        });
    }
 
    var obj = {"currentPage": currentPage, "itemsPerPage": itemsPerPage};
    $scope.getListNames();
    $scope.getList(obj);
 
    $scope.getItems = function($query) {
        return hosts.filter(function(host) {
            return host.toLowerCase().indexOf($query.toLowerCase()) != -1;
        });
    }
    
    $scope.checkedRoutes = function () {
        $scope.routes.forEach(function(route){
            route.checked = $scope.allCheck;
        })
    }
 
    $scope.detailRoute = function (route) {
        $scope.resetData();
        $rootScope.loadingDetail = true;
        var routePromise = routeServices.getRoute(route.guid);
        routePromise.success(function(data) {
            $rootScope.loadingDetail = false;
 
            $scope.detail = data;
            if(data.created)
                $scope.detail.created = moment(data.created).format("YYYY.MM.DD HH:mm");
            if(data.updated)
                $scope.detail.updated = moment(data.updated).format("YYYY.MM.DD HH:mm");
        })
        routePromise.error(function (data) {
            $rootScope.loadingDetail = false;
            cloudfoundry.alert(data);
        });
    }
 
    $scope.resetData = function () {
        $scope.detail = null;
    }
});
