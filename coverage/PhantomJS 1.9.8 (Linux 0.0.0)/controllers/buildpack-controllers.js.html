'use strict';
 
var buildpackControllers = angular.module('buildpack.controllers', []);
 
buildpackControllers.controller('BuildpackManageController', function ($rootScope, $scope, cloudfoundry, buildpackServices, applicationServices, commonUtil, $translate) {
 
    $scope.titles = commonUtil.getTitleLinks();
    $scope.adminTitles = commonUtil.getAdminTitles();
 
    $scope.selections = ['true', 'false'];
 
    var currentPage = 1; 
    var itemsPerPage = 10;
    var buildpacks = [];
 
    var positionOptions = null;
 
    $scope.getList = function (obj) {
        var condition = '';
 
        if(obj == undefined) {
            obj = {"currentPage":$scope.currentPage, "itemsPerPage":$scope.itemsPerPage};
        }
        var requestPage = obj.currentPage-1;
 
        if(($scope.searchField == undefined || $scope.searchField == '') && jQuery('#searchText').text() != '' && jQuery('#searchText').text().trim() != $translate.instant('buildpack_name_search')) {
            cloudfoundry.alert($translate.instant('mi_search_predictive_func'));
            return;
        }
        if($scope.searchField != undefined && (typeof $scope.searchField == 'object' && $scope.searchField.length > 0)) {
            $scope.currentPage = 1;
            obj.currentPage = 1;
            requestPage = 0;
            condition = "name:" + $scope.searchField[0].name;
        }
 
        $rootScope.loading = true;
        var buildpackPromise = buildpackServices.listBuildpacks(obj.itemsPerPage, requestPage, condition);
        buildpackPromise.success(function (data) {
            $rootScope.loading = false;
 
            $scope.totalItems = data.totalElements;
            $scope.currentPage = obj.currentPage;
            $scope.itemsPerPage = obj.itemsPerPage;
            $scope.totalPages = Math.ceil(data.totalElements / obj.itemsPerPage); 
 
            $scope.buildpacks = data.content;
 
            // position
            $scope.position = 1;
            positionOptions = new Array($scope.totalItems);
            for(var i = 0; i < $scope.totalItems; i++){
                positionOptions[i] = i+1;
            }
            $scope.positionOptions = positionOptions;
 
            $scope.createPositionOptions = angular.copy(positionOptions); 
            $scope.createPositionOptions.push($scope.totalItems+1);
        });
        buildpackPromise.error(function (data) {
            $rootScope.loading = false;
            cloudfoundry.alert(data);
        });
    }
 
    $scope.getListNames = function () {
        var buildpackPromise = buildpackServices.listAllBuildpacks();
        buildpackPromise.success(function(data) {
            buildpacks = [];
            angular.forEach(data, function(buildpack) {
                buildpacks.push({'name': buildpack.name});
            });
        })
        buildpackPromise.error(function (data) {
            cloudfoundry.alert(data);
        });
    }
 
    var obj = {"currentPage": currentPage, "itemsPerPage": itemsPerPage};
    $scope.getListNames();
    $scope.getList(obj);
 
    $scope.getItems = function($query) {
        return buildpacks.filter(function (buildpack) {
            return buildpack.name.toLowerCase().indexOf($query.toLowerCase()) != -1;
        });
    }
 
    $scope.checkedBuildpacks = function () {
        $scope.buildpacks.forEach(function(buildpack){
            buildpack.checked = $scope.allCheck;
        });
    }
 
    $scope.countCheckedBuildpacks = function () {
        $scope.deleteMessage = '';
        $scope.count = 0;
 
        if(!$scope.buildpacks) {
            return;
        }
        
        var count = 0;
        $scope.buildpacks.forEach(function (buildpack) {
            if(buildpack.checked) {
                count++;
            }
        });
 
        if(count != 0) {
            $scope.deleteMessage = $translate.instant('mq_delete_buildpack_count');
            $scope.count = count;
        }
    }
 
    $scope.detailBuildpack = function (buildpack) {
        $scope.resetData();
        $rootScope.loadingDetail = true;
        $scope.applications = [];
        var buildpackPromise = buildpackServices.getBuildpack(buildpack.guid);
        buildpackPromise.success(function(data) {
            $rootScope.loadingDetail = false;
 
            $scope.detail = data;
            $scope.updateFile = { "name" : data.filename };
            if(data.created)
                $scope.detail.created = moment(data.created).format("YYYY.MM.DD HH:mm");
            if(data.updated)
                $scope.detail.updated = moment(data.updated).format("YYYY.MM.DD HH:mm");
        })
        buildpackPromise.error(function (data) {
            $rootScope.loadingDetail = false;
            cloudfoundry.alert(data);
        });
    }
 
    $scope.select = function (idx, item) {
        if(idx==0) {
            $scope.enabled = true;
            item = true;
        } else {
            $scope.enabled = false;
            item = false;
        }
    }
    
    $scope.selectPosition = function (idx) {
        $scope.poition = idx+1;
    }
 
    $scope.getApplications = function (buildpackName) {
        var itemsPerPage = 100;
        var requestPage = 0;
        $rootScope.loadingDetail = true;
        var applicationPromise = applicationServices.listApplicationsByBuildpack(itemsPerPage, requestPage, buildpackName);
        applicationPromise.success(function (data) {
            $rootScope.loadingDetail = false;
            $scope.applications = data;
        });
        applicationPromise.error(function (data) {
            $rootScope.loading = false;
            cloudfoundry.alert(data);
        });
    }
 
    $scope.createBuildpack = function () {
        if($scope.name == undefined || $scope.name == ''){
            cloudfoundry.alert($translate.instant('mi_dont_input_buildpack_name'));
            return;
        }
        if($scope.file == undefined || $scope.file == ''){
            cloudfoundry.alert($translate.instant('mi_dont_select_file'));
            return;
        }
        if($scope.selPosition == undefined || $scope.selPosition == ''){
            cloudfoundry.alert($translate.instant('mi_dont_select_priority'));
            return false;
        }
        if($scope.enabled == undefined || $scope.enabled == ''){
            cloudfoundry.alert($translate.instant('mi_dont_select_use_boolean'));
            return;
        }
        var data = {};
        if($scope.version != undefined && $scope.version != null)
            data['name'] = $scope.name + '_buildpack-v' + $scope.version;
        else
            data['name'] = $scope.name + '_buildpack';
        data['position'] = $scope.selPosition;
        data['enabled'] = $scope.enabled;
        data['file'] = $scope.file;
 
        if($scope.file != null) {
            buildpackServices.createBuildpack(data, $scope);
        }
    }
 
    $scope.updateBuildpack = function () {
        if($scope.updateFile == undefined || $scope.updateFile == ''){
            cloudfoundry.alert($translate.instant('mi_dont_select_file'));
            return;
        }
        if($scope.detail.position == undefined || $scope.detail.position == ''){
            cloudfoundry.alert($translate.instant('mi_dont_select_priority'));
            return false;
        }
        if($scope.detail.enabled == undefined || $scope.detail.enabled == ''){
            cloudfoundry.alert($translate.instant('mi_dont_select_use_boolean'));
            return;
        }
        var data = {};
        data['guid'] = $scope.detail.guid;
        data['name'] = $scope.detail.name;
        data['position'] = $scope.detail.position;
        data['enabled'] = $scope.detail.enabled;
        data['locked'] = $scope.detail.locked;
        data['file'] = $scope.updateFile;
 
        buildpackServices.updateBuildpack(data, $scope);
    }
 
    $scope.deleteBuildpacks = function () {
        $rootScope.rootLoading = true;
        $scope.buildpacks.forEach(function(buildpack){
            if(buildpack.checked === true){
                var buildpackPromise = buildpackServices.deleteBuildpack(buildpack.guid);
                buildpackPromise.success(function (data) {
                    $rootScope.rootLoading = false;
                    cloudfoundry.success($translate.instant('mi_delete_buildpack'));
                    $scope.totalItems = $scope.totalItems - 1;
                    var totalPages = 0;
 
                    if($scope.totalItems > $scope.itemsPerPage)
                        totalPages = Math.ceil($scope.totalItems/$scope.itemsPerPage);
                    else
                        totalPages = 1;
 
                    if($scope.currentPage > totalPages)
                        obj.currentPage = $scope.currentPage - 1;
                    else
                        obj.currentPage = $scope.currentPage;
 
                    obj.itemsPerPage = $scope.itemsPerPage;
                    $scope.getList(obj);
                    $scope.getListNames();
                });
                buildpackPromise.error(function (data) {
                    $rootScope.rootLoading = false;
                    cloudfoundry.alert(data);
                });
            }
        });
    }
 
    $scope.resetBuildpack = function () {
        $scope.updateFile = null;
    }
    
    $scope.resetForm = function () {
        $scope.name = null;
        $scope.version = null;
        $scope.file = null;
        $scope.selPosition = null;
        $('.rads').prop('checked', false);
        $scope.addForm.name.$pristine = true;
        $scope.addForm.name.$touched = false;
        $scope.addForm.version.$pristine = true;
        $scope.addForm.version.$touched = false;
    }
 
    $scope.resetData = function () {
        $scope.detail = null;
        $scope.updateFile = null;
        $scope.applications = null;
    }
});
