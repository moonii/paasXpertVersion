'use strict';
 
var systemControllers = angular.module('system.controllers', []);
 
systemControllers.controller('VirtualMachineManageController', function ($rootScope, $scope, cloudfoundry, commonUtil, cache, vmServices, userServices) {
    $scope.cache = cache;
 
    $scope.titles = commonUtil.getTitleLinks();
    $scope.systemTitles = commonUtil.getSystemTitles();
 
    var regionId = 0;
    $scope.getVMs = function() {
        $rootScope.loading = true;
        var userPromise = userServices.getPortalUser(cache.getUser().id);
        userPromise.success(function (data) {
            regionId = data.defaultRegion.id;
 
            var vmPromise = vmServices.listAllDeployments(regionId);
            vmPromise.success(function (data) {
                $rootScope.loading = false;
                $scope.deployments = data;
                $scope.selDeploymentName = $scope.deployments[0].name;
                $scope.listVMsByDeployment($scope.selDeploymentName);
            });
            vmPromise.error(function (data) {
                $rootScope.loading = false;
                cloudfoundry.alert(data);
            });
        });
        userPromise.error(function (data) {
            $rootScope.loading = false;
            cloudfoundry.alert(data);
        });
    }
    $scope.getVMs();
 
    $scope.getList = function() {
        $rootScope.loading = true;
        var vmPromise = vmServices.listAllVMs(regionId);
        vmPromise.success(function (data) {
            $rootScope.loading = false;
            $scope.virtualMachines = data;
        });
        vmPromise.error(function (data) {
            cloudfoundry.alert(data);
        });
    }
 
    $scope.listVMsByDeployment = function(deploymentName) {
        $rootScope.loading = true;
        var vmPromise = vmServices.listVMsByDeployment(regionId, deploymentName);
        vmPromise.success(function (data) {
            $rootScope.loading = false;
            $scope.virtualMachines = data;
        });
        vmPromise.error(function (data) {
            $rootScope.loading = false;
            cloudfoundry.alert(data);
        });
    }
 
    $scope.detailVirtualMachine = function(virtualMachine) {
        $scope.resetData();
        $scope.detail = virtualMachine;
    }
 
    $scope.selectDeployment = function() {
        if($scope.selDeployment != undefined || $scope.selDeployment != null) {
            $scope.selDeploymentName = $scope.selDeployment.name;
            $scope.listVMsByDeployment($scope.selDeploymentName);
        }
    }
 
    $scope.resetData = function () {
        $scope.detail = null;
    }
});
