                if (evt.data.toString().indexOf("Ready") > -1) {
                    $scope.$apply(function() {
                        $scope.jstreeInit(1);
                    });

                    var data = evt.data;
                    if(selNodeEl == undefined || selNodeEl == 'tree') {

                        var begin_text = 'vcap';
                        // docker app의 경우 home directory가 다를 수 있기 때문에 /를 home으로 인식하도록 수정
                        // buildpack이 null인 경우를 docker app으로 판단함
                        if($scope.application.buildpack == null) {
                            if(data.indexOf('DirectoryInfo') > -1) {
                                var jsonMsg = JSON.parse(data);
                                var jsonDir = jsonMsg.DirectoryInfo;
                                if(JSON.stringify(jsonDir[0]).indexOf('/home/vcap') > -1) {
                                    data = data.replace(/\/home\/vcap/g, "");
                                }
                            }
                            begin_home = '/';
                            begin_text = '/';
                        }
                        setTree(data, begin_home, begin_text);
                    } else if(selNodeEl == 'file') {
                        setView(data);
                    }
                }
            }

                $scope.$apply(function() {
                    $scope.loadingApplicationInstance = false;
                    $scope.sshConnected = true;
                    $scope.websocketStatus = 'Connected';
                });
            }
            conn.onclose = function(evt) {
                $scope.$apply(function() {

                        "opened":true,
                        "disabled":false,
                        "selected":false
                    },
                    "li_attr":{
                        "isLeaf":false
                    },
                    "children":true
                }];
                $scope.msg = 'ls';
            } else {
                var msg = data.replace(/\\r/g, "");
                var jsonMsg = JSON.parse(msg);
                var jsonDir = jsonMsg.DirectoryInfo;
                angular.forEach($scope.treeModel, function(node) {
                    if(node.id == parent) {
                        node['state'] = {"opened":true, "selected":true};
                        delete node.children;
                    } else {
                        if(node.state && node.state.selected) node.state.selected = false;
                    }
                });
                for(var j in jsonDir) {
                    if(j > 1) { // ls 명령에서 1,2번째는 '.'과 '..'이기 때문에 제외함
                        jsonDir[j]['parent'] = parent;
                        var isLeaf = true;
                        if(jsonDir[j].children == true) isLeaf = false;
                        jsonDir[j]['li_attr'] = {"isLeaf":isLeaf};
                        $scope.treeModel.push(jsonDir[j]);
                    }
                }
            }
        });
    }

    $scope.getUpTime = function(interval) {
        // https://msdn.microsoft.com/ko-kr/library/ee532932(v=vs.94).aspx
        var msecPerMinute = 1000 * 60;
        var msecPerHour = msecPerMinute * 60;
        var msecPerDay = msecPerHour * 24;
 
        interval = interval * 1000;
 
        var days = Math.floor(interval / msecPerDay );
        interval = interval - (days * msecPerDay );
 
        var hours = Math.floor(interval / msecPerHour );
        interval = interval - (hours * msecPerHour );
 
        var minutes = Math.floor(interval / msecPerMinute );
        interval = interval - (minutes * msecPerMinute );
 
        var seconds = Math.floor(interval / 1000 );
 
        //console.log(days + " days, " + hours + " hours, " + minutes + " minutes, " + seconds + " seconds.");
        return days + $translate.instant('day') + ' ' + hours + $translate.instant('hour') + ' ' + minutes + $translate.instant('minute');
