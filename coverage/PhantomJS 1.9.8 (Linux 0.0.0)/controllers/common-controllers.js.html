commonControllers.controller('AdminMainController', function ($rootScope, $scope, $location, $route, cloudfoundry, $cookies, $translate) {
    // 로그인 되지 않은 상태에서 허용할 페이지 설정
    var paths = ['/login', '/pw-request', '/reset-password', '/verify'];
    var compare = $location.path();
 
    if (!cloudfoundry.isAuthenticated()) {
        if(compare.indexOf('/reset-password') > -1) compare = '/reset-password';
        if(compare.indexOf('/verify') > -1) compare = '/verify';
 
        if ($.inArray(compare, paths) === -1) {
            $location.path('/login');
            return;
        }
    }
 
    if ($location.path() === '/' || $location.path().length === 0 || cloudfoundry.isAuthenticated() && ($.inArray(compare, paths) > -1)) {
        $location.path(introduction);
    } else {
        $route.reload();
    }
});
 
commonControllers.controller('NavigationController', function ($window, $scope, $location, $http, cloudfoundry, cache, regionServices, userServices, $translate, $cookies) {
    $scope.cache = cache;
 
    $('body').find('a').removeClass('on');
    $('.group_setting').hide();
    var id = $location.path();
    id = id.substring(1, $location.path().length).replace(/\/.*/,'');
 
    jQuery('#'+id).addClass('on').parent('li').addClass('on').parents('ul').prev('a').addClass('on');
    jQuery('#gnb > div> ul > li').each(function() {
        if($(this).find('li').hasClass('on')) {
            $(this).find('> a').addClass('on');
            $(this).children('.group_setting').show();
        };
    });
 
	jQuery('#gnb_scroll').css('height',jQuery(window).height()-294);
 
	gnbScroll = new IScroll('#gnb_scroll', {
		mouseWheel: true,
		scrollbars: true
	});
	gnbScroll.refresh();
 
    $scope.logout = function () {
        $http.get(serverUrl + '/api/core/ping')
            .then(
                function(response){
                    cloudfoundry.logout();
                    $location.path('/login');
                },
                function(response){
                    cloudfoundry.alert(response);
                    $location.path('/login');
                }
            );
    }
 
    var quotaPromise = cloudfoundry.getAllQuotas();
    quotaPromise.success(function (data, status, headers) {
        cache.storeQuotas(data);
    });
    quotaPromise.error(function (data) {
        cloudfoundry.alert(data);
    });
 
    $scope.isAuthenticated = cloudfoundry.isAuthenticated();
    if (cloudfoundry.isAuthenticated()) {
        $scope.user = cache.getUser();
        getRegion();
    }
    else {
        cloudfoundry.alert($translate.instant('mi_invalid_session'));
        $scope.logout();
    }
    // Region
    function getRegion() {
        var regionPromise = cloudfoundry.getRegions(cache.getUser().id);
        regionPromise.success(function (data) {
            $scope.regions = data;
            // Default Region
            var userPromise = userServices.getPortalUser(cache.getUser().id);
            userPromise.success(function (data) {
                var defaultRegionId = data.defaultRegion.id;
                angular.forEach($scope.regions, function(region) {
                    if(region.id == defaultRegionId) {
                        $scope.selRegion = region;
                    }
                });
            });
            userPromise.error(function (data) {
                data.message = data.message + '<br/>[getPortalUser]';
                cloudfoundry.alert(data);
            });
        });
        regionPromise.error(function (data) {
            data.message = data.message + '<br/>[getRegions]';
            cloudfoundry.alert(data);
        });
    }
 
    $scope.changeRegion = function(selRegion) {
        var updatePromise = regionServices.updateDefaultRegion(cache.getUser().id, selRegion.id);
        updatePromise.success(function (data) {
            var updatePromise = regionServices.updateUserDefaultRegion(cache.getUser().id);
            updatePromise.success(function () {
                cloudfoundry.success($translate.instant('mi_change_region'));
                $window.location.reload();
            });
            updatePromise.error(function (data) {
               cloudfoundry.alert(data);
            });
        });
        updatePromise.error(function (data) {
            cloudfoundry.alert(data);
        });
    }
 
    $scope.selLanguage = $cookies.get("translateKey");
 
    $scope.changeLanguage = function (key) {
        $translate.use(key);
        $cookies.put("translateKey", key);
        $scope.selLanguage = key;
    }
})

// 모든 Controller에서 공통함수를 사용하기 위하여 아래 RootController를 상속받아 사용하도록 처리
function RootController($scope, $rootScope) {
    $scope.go = function (menu) {
        $rootScope.chooseMenu = menu;
    }
    var id = '#'+$rootScope.chooseMenu;
    $(id).addClass('list-group-item-active');
    $(id).parents().addClass('in');
    var parent = $(id).parent();
    parent.prev().attr('aria-expanded','true');
    //parent.siblings('a[data-target="#'+parent.attr('id')+'"]').attr('aria-expanded','true'); // 위에꺼랑 같음
}
 
// stackable modal. modal을 여러 개 중첩되어 띄울 경우 먼저 띄운 modal은 흐리게 처리
$(document).on('show.bs.modal', '.modal', function (event) {
    var zIndex = 1040 + (10 * $('.modal:visible').length);
    $(this).css('z-index', zIndex);
    setTimeout(function() {
        $('.modal-backdrop').not('.modal-stack').css('z-index', zIndex - 1).addClass('modal-stack');
    }, 0);
});
