        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/placement_groups?size='+size+'&page='+page+'&condition='+condition, 'GET'));
    }
 
    cellServices.getPlacementGroup = function(guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/placement_groups/'+guid, 'GET'));
    }

        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/placement_groups', 'POST', data));

    }
 
    cellServices.deletePlacementGroup = function(guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/placement_groups/'+guid, 'DELETE'));
    }

        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/cells/diego', 'GET'));
