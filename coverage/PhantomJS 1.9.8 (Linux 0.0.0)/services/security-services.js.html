'use strict';
 
var securityServices = angular.module('security.services', ['common.services']);
 
securityServices.factory('securityServices', function ($http, cache, $q, $compile, cloudfoundry) {
 
    var securityServices = {};
 
    securityServices.listAllSecurityGroups = function () {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/security_groups/all', 'GET'));
    }
    
    securityServices.listSecurityGroups = function (size, page, condition) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/security_groups?size='+size+'&page='+page+'&condition='+condition, 'GET'));
    }
 
    securityServices.getSecurityGroup = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/security_groups/' + guid, 'GET'));
    }
    
    securityServices.createSecurityGroup = function (data) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/security_groups', 'POST', data));
    }
 
    securityServices.updateSecurityGroup = function (data) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/security_groups/' + data.guid, 'PUT', data));
    }
 
    securityServices.deleteSecurityGroup = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/security_groups/' + guid, 'DELETE'));
    }
 
    securityServices.bindStagingSecurityGroup = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/security_groups/staging/' + guid, 'PUT'));  
    }
 
    securityServices.unbindStagingSecurityGroup = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/security_groups/staging/' + guid, 'DELETE'));
    }
 
    securityServices.bindRunningSecurityGroup = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/security_groups/running/' + guid, 'PUT'));
    }
 
    securityServices.unbindRunningSecurityGroup = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/security_groups/running/' + guid, 'DELETE'));
    }
    
    return securityServices;
 
});
