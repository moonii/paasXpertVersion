'use strict';
 
/**
 * See http://stackoverflow.com/questions/25173132/angularjs-is-not-sending-x-xsrf-token
 *
 * @ngdoc service
 * @name angularSampleApp.XSRFInterceptor
 * @description
 * # XSRFInterceptor
 * Factory in the angularSampleApp.
 */
angular.module('portaladmin')
  .factory('XSRFInterceptor', function ($rootScope, $location, $q, $log) {
 
    //var XSRFInterceptor = {
    //
    //  request: function(config) {
    //
    //    var token = $cookies.get('XSRF-TOKEN');
    //
    //    if (token) {
    //      config.headers['X-XSRF-TOKEN'] = token;
    //      $log.info("X-XSRF-TOKEN: " + token);
    //    }
    //
    //    return config;
    //  }
    //};
 
    var xsrfToken;
 
    var XSRFInterceptor = {
 
      request: function(config) {
 
        if (xsrfToken) {
            // webproxy에서 header에 X-XSRF-TOKEN 값이 있으면 접속할 수 없기 때문에 분기처리
            if((config.url.indexOf(sshProxy) < 0) && (config.xhrFields == undefined)) {
                config.headers['X-XSRF-TOKEN'] = xsrfToken;
            }
          //$log.info("X-XSRF-TOKEN being sent to server: " + xsrfToken);
        }
 
        return config;
      },
 
      response: function(response) {
 
        var newToken = response.headers('X-XSRF-TOKEN');
 
        if (newToken) {
          xsrfToken = newToken;
          //$log.info("X-XSRF-TOKEN received from server: " + xsrfToken);
        }
 
        return response;
      },
        responseError: function(rejection) {
            $rootScope.loading = false;
            if(rejection.status == 403) {
                $location.path('/login').search({status: 403});
                return false;
            }
            // webproxy에 연결이 안 되어 있을 때도 status가 -1이니 분기처리
            if (rejection.status == -1 && rejection.config.url.indexOf(sshProxy) < 0) {
                $location.path('/login').search({status: -1});
                return false;
            }
            return $q.reject(rejection);
        }
    };
 
    return XSRFInterceptor;
  });
