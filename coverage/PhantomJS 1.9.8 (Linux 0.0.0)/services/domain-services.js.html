'use strict';
 
var domainServices = angular.module('domain.services', ['common.services']);
 
domainServices.factory('sharedDomainServices', function ($http, cache, $q, $compile, cloudfoundry) {
 
    var sharedDomainServices = {};
 
    sharedDomainServices.listAllSharedDomains = function () {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/shared_domains/all', 'GET'));
    }
 
    sharedDomainServices.listSharedDomains = function (size, page, condition) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/shared_domains?size='+size+'&page='+page+'&condition='+condition, 'GET'));
    }
 
    sharedDomainServices.getSharedDomain = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/shared_domains/' + guid, 'GET'));
    }
 
    sharedDomainServices.createSharedDomain = function (data) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/shared_domains', 'POST', data));
    }
 
    sharedDomainServices.updateSharedDomain = function (data) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/shared_domains/' + data.guid + '?name=' + data.name, 'PUT'));
    }
 
    sharedDomainServices.deleteSharedDomain = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/shared_domains/' + guid, 'DELETE'));
    }
 
    return sharedDomainServices;
 
});
