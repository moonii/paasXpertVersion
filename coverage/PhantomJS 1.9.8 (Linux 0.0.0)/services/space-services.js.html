'use strict';
 
var spaceServices = angular.module('space.services', ['common.services']);
 
spaceServices.factory('spaceServices', function (cloudfoundry) {
 
    var spaceServices = {};
 
    spaceServices.listAllSpaces = function () {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/all', 'GET'));
    }
    
    spaceServices.listSpaces = function (size, page, condition) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces?size=' + size + '&page=' + page + '&condition=' + condition, 'GET'));
    }
 
    spaceServices.listAllOrgSpaceQuotas = function(organizationId) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/space_quotas/org/' + organizationId + '/all', 'GET'));
    }
 
    spaceServices.getSpace = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/' + guid, 'GET'));
    }
 
    spaceServices.getOrgQuotaUsage = function (organizationId, quotaMappingId) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/organizations/' + organizationId + '/quota/' + quotaMappingId + '/usage', 'GET'));
    }
 
    spaceServices.getSpaceQuotaUsage = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/' + guid + '/quota_summary', 'GET'));
    }
 
    spaceServices.createSpace = function (body) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/norole', 'POST', body));
    }
 
    spaceServices.updateSpace = function (spaceId, body) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/' + spaceId, 'PATCH', body));
    }
 
    spaceServices.setSpaceQuota = function (spaceId, spaceQuotaGuid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/' + spaceId + '/space_quotas/' + spaceQuotaGuid, 'PUT'));
    }
 
    spaceServices.deleteSpace = function (guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/' + guid, 'DELETE'));
    }
    
    return spaceServices;
    
});

        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/space_quotas/all/noorg', 'GET'));

        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/space_quotas/noorg?size='+size+'&page='+page, 'GET'));

        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/space_quotas/org/' + organizationId + '/detail' + '?size='+size+'&page='+page, 'GET'));
    }
 
    spaceQuotaServices.getSpaceQuotaDetail = function(guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/spaces/space_quotas/detail/' + guid, 'GET'));
    }
 
    spaceQuotaServices.getSpacesBySpaceQuotaDefinition = function (guid) {

            '/' + guid, 'GET'));

});
