'use strict';
 
var loginService = angular.module('login.services', ['common.services']);
 
loginService.factory('loginService', function ($http, $q, cache, cloudfoundry, userServices) {
 
    var loginService = {};
 
    loginService.authenticate = function(credentials) {
        cache.clear();
 
        var promise = $http({
            method: 'POST',
            url: serverUrl + '/login',
            headers: {
                'Content-Type': 'application/x-www-form-urlencoded'
            },
            data: $.param(credentials)
        });
        return cloudfoundry.retrieveResource(promise, function (authenticationDetails) {
            cache.storeUser(authenticationDetails)
        });
    }
 
    loginService.requestResetEmail = function(email) {
        var promise = $http({
            method: 'POST',
            url: serverUrl + '/api/core/forgot-admin-password',
            headers: {'Content-Type': 'application/x-www-form-urlencoded;charset=utf-8', 'Accept': 'application/json;charset=utf-8'},
            data: $.param({ 'email': email })
        });
        return cloudfoundry.retrieveResource(promise);
    }
 
    loginService.resetPassword = function(body) {
        var promise = $http({
            method: 'POST',
            url: serverUrl + '/api/core/users/' + body.verifyCode + '/reset-password',
            headers: {'Content-Type': 'application/x-www-form-urlencoded;charset=utf-8', 'Accept': 'application/json;charset=utf-8'},
            data: $.param({ 'newPassword': body.password})
        });
        return cloudfoundry.retrieveResource(promise);
    }
 
    loginService.signup = function(form) {
        var promise = $http({
            method: 'POST',
            url: serverUrl + '/api/core/signup',
            headers: {'Content-Type': 'application/json;charset=utf-8', 'Accept': 'application/json;charset=utf-8'},
            data: {'email': form.email,
                'password': form.password,
                'name': form.name
            }
        });
        return cloudfoundry.retrieveResource(promise);
    }
 
    loginService.resendVerificationMail = function(user) {
        var promise = $http({
            method: 'GET',
            url: serverUrl + '/api/core/users/' + user + '/resend-verification-mail',
            headers: {'Content-Type': 'application/json;charset=utf-8', 'Accept': 'application/json;charset=utf-8'},
        });
        return cloudfoundry.retrieveResource(promise);
    }
 
    loginService.verifyUser = function(verifyCode) {
        ///api/core/users/{verificationCode}/verify
        var promise = $http({
            method: 'POST',
            url: serverUrl + '/api/core/users/' + verifyCode + '/verify',
            headers: {'Content-Type': 'application/json', 'Accept': 'application/json;charset=utf-8'}
        });
        return cloudfoundry.retrieveResource(promise);
    }
 
    return loginService;
 
});
