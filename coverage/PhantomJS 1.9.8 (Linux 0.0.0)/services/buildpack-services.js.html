'use strict';
 
var buildpackServices = angular.module('buildpack.services', ['common.services']);
 
buildpackServices.factory('buildpackServices', function ($rootScope, $http, cache, $q, $compile, cloudfoundry, Upload, $translate) {
 
	var buildpackServices = {};
        
    buildpackServices.listAllBuildpacks = function() {
		return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/buildpacks/all', 'GET'));
	}
    
    buildpackServices.listBuildpacks = function(size, page, condition) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/buildpacks?size='+size+'&page='+page+'&condition='+condition, 'GET'));
	}
    
    buildpackServices.getBuildpack = function(guid) {
		return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/buildpacks/'+guid, 'GET'));
	}
    
    buildpackServices.createBuildpack = function(data, scope) {
		$rootScope.rootLoading = true;
		$http.get(serverUrl + '/api/core/ping')
			.then(
				function(response){
					Upload.upload({
						method: 'POST',
						url: serverUrl + context + apiVersion + '/buildpacks',
						headers: {
							'Content-Type': 'multipart/form-data'
						},
						data: data
					}).then(function (resp) {
						if(resp.data){
							$rootScope.rootLoading = false;
							cloudfoundry.success($translate.instant('mi_create_buildpack'));
							scope.totalItems = scope.totalItems + 1;
							var totalPages = 0;
 
							if(scope.totalItems > scope.itemsPerPage)
								totalPages = Math.ceil(scope.totalItems/scope.itemsPerPage);
							else
								totalPages = 1;
 
							var obj = {};
							obj.currentPage = totalPages;
							obj.itemsPerPage = scope.itemsPerPage;
							scope.getListNames();
							scope.getList(obj);
						}else{
							$rootScope.rootLoading = false;
							cloudfoundry.alert($translate.instant('mi_error'));
						}
					}, function (xhr, status, error) {
						$rootScope.rootLoading = false;
						cloudfoundry.alert(xhr.data.message);
					}, function (evt) {
						var progressPercentage = parseInt(100.0 * evt.loaded / evt.total);
						if(evt.config.data.file){
							console.log('progress: ' + progressPercentage + '% ' + evt.config.data.file.name);
						}
					});
				},
				function(response){
					scope.$root.loading = false;
					cloudfoundry.alert(response);
				}
			);
	}
 
    buildpackServices.updateBuildpack = function(data, scope) {
		$rootScope.rootLoading = true;
		$http.get(serverUrl + '/api/core/ping')
			.then(
				function(response){
					Upload.upload({
						method: 'POST',
						url: serverUrl + context + apiVersion + '/buildpacks/' + data.guid,
						headers: {
							'Content-Type': 'multipart/form-data'
						},
						data: data
					}).then(function (resp) {
						if(resp.data){
							$rootScope.rootLoading = false;
							cloudfoundry.success($translate.instant('mi_edit_buildpack'));
							var obj = {"currentPage": scope.currentPage, "itemsPerPage": scope.itemsPerPage, "condition": ""};
							scope.getList(obj);
						}else{
							cloudfoundry.alert($translate.instant('mi_error'));
						}
					}, function (xhr, status, error) {
						cloudfoundry.alert('Error!! ' + xhr.data.message);
					}, function (evt) {
						var progressPercentage = parseInt(100.0 * evt.loaded / evt.total);
						if(evt.config.data.file){
							console.log('progress: ' + progressPercentage + '% ' + evt.config.data.file.name);
						}
					});
				},
				function(response){
					scope.$root.loading = false;
					cloudfoundry.alert(response);
				}
			);
	}
    
    buildpackServices.deleteBuildpack = function(guid) {
		return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/buildpacks/'+guid, 'DELETE'));
	}
 
	return buildpackServices;
	
});
