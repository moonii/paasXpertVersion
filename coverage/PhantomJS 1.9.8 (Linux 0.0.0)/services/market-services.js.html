'use strict';
 
var appItemServices = angular.module('appItem.services', ['common.services']);
 
appItemServices.factory('appItemServices', function ($http, cache, cloudfoundry) {
 
    var appItemServices = {};
 
    appItemServices.listAllAppItems = function() {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/appMarkets/appItem/all', 'GET'));
    }
 
    appItemServices.listAppItems = function(size, page, category, name) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/appMarkets?size='+size+'&page='+page+'&category='+category+'&name='+name, 'GET'));
    }
 
    appItemServices.listAppItemsByState = function(size, page, category, state) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/appMarkets/state?size='+size+'&page='+page+'&category='+category+'&state='+state, 'GET'));
    }
 
    appItemServices.listAppItemsByOwner = function(size, page, category, owner) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/appMarkets/owner?size='+size+'&page='+page+'&category='+category+'&owner='+owner, 'GET'));
    }
 
    appItemServices.getAppItem = function(guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/appMarkets/appItem/' + guid, 'GET'));
    }
 
    appItemServices.getAppItemDownloadUserByAppItemGuid = function(guid) {
        return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/appMarkets/downapps/appItem/' + guid, 'GET'));
    }
 
    return appItemServices;
});
