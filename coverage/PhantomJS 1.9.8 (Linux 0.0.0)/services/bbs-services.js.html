'use strict';
 
var bbsServices = angular.module('bbs.services', ['common.services']);
 
bbsServices.factory('boardService', function ($rootScope, $http, cache, $q, $compile, cloudfoundry) {
 
	var boardService = {};
 
	boardService.listAllBoard = function() {
		return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/board/all', 'GET'));
	};
 
	boardService.listBoard = function(size, page, type, name) {
		return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/board?size='+size+'&page='+page+'&type='+type+'&name='+name, 'GET'));
	};
 
	boardService.getBoard = function(guid) {
		return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/board/'+guid, 'GET'));
	};
 
	boardService.addBoard = function(body) {
		return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/board', 'POST', body));
	};
 
	boardService.deleteBoard = function(guid) {
		return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/board/'+guid, 'DELETE'));
	};
 
	boardService.updateBoard = function(guid, body) {
		return cloudfoundry.retrieveResource(cloudfoundry.resourcePromise(serverUrl + context + apiVersion + '/board/'+guid, 'PATCH', body));
	};
    
	return boardService;
});
